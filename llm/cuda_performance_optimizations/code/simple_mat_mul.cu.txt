// simple_mat_mul.cu
// nvcc simple_mat_mul.cu -o simple_mat_mul -lcublas

#include <cstdio>
#include <cublas_v2.h>
#include <cuda_runtime.h>

int main() {
    cublasHandle_t handle;
    cublasCreate(&handle);

    const int N = 2;
    float h_A[N * N] = {1.0f, 3.0f,
                        2.0f, 4.0f};
    float h_B[N * N] = {5.0f, 7.0f,
                        6.0f, 8.0f};
    float h_C[N * N] = {0};

    float *d_A, *d_B, *d_C;
    cudaMalloc((void**)&d_A, N * N * sizeof(float));
    cudaMalloc((void**)&d_B, N * N * sizeof(float));
    cudaMalloc((void**)&d_C, N * N * sizeof(float));

    cudaMemcpy(d_A, h_A, N * N * sizeof(float), cudaMemcpyHostToDevice);
    cudaMemcpy(d_B, h_B, N * N * sizeof(float), cudaMemcpyHostToDevice);

    const float alpha = 1.0f;
    const float beta = 0.0f;

    // cuBLAS assumes column-major layout, so row-major A x B becomes B x A here.
    cublasSgemm(handle,
                CUBLAS_OP_N, CUBLAS_OP_N,
                N, N, N,
                &alpha,
                d_B, N,
                d_A, N,
                &beta,
                d_C, N);

    cudaMemcpy(h_C, d_C, N * N * sizeof(float), cudaMemcpyDeviceToHost);

    printf("Result matrix C:\n");
    for (int i = 0; i < N; ++i) {
        printf("%f %f\n", h_C[i], h_C[i + N]);
    }

    cudaFree(d_A);
    cudaFree(d_B);
    cudaFree(d_C);
    cublasDestroy(handle);
    return 0;
}
